
1. printf("%03d\n", m); // 当m不超过三位时，输出占三位，前补0；当m超过三位，原样输出

// 明明是整数，却要以浮点数方式输出；明明是浮点数，却要以整数方式输出
// 对输出结果的探究涉及到整数和浮点数的编码
2. printf("%.1f\n", 8/5); // 0.0
3. printf("%d\n", 8.0/5.0); // 256

//在C99中，double的输出必须用%f，而输入需要用%lf，但是在C89和C++中都不必如此，输入输出可以都用%lf
4. double a;
5. scanf("%lf", &a);
6. printf("%f\n", a);
